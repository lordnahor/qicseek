71  /***
72   * ProxyFilter decides where an HTTP request should be handled. If one peers
73   * owns the file mentioned in the request, the request goes to that peer. This
74   * could be the local peer. If no one owns the file, then the request goes to
75   * the local peer, so the local peer might create the file This is a request
76   * filter. A response filter does not fulfill our needs. Simple redirect does
77   * not work with WebDAV methods other than GET.
78   * 
79   * @author Jie Ren
80   */
81  public class ProxyFilter implements Filter, PeerOwner {

160     /***
161      * Perform initialization after the WebdavServlet is initialized
162      */
163     public void initAfterServlet() {
164         // Get Jetty server to shutdown
165         jettyServer = (HttpServer) filterConfig.getServletContext().getAttribute(
166                 LevelHandler.JETTY_NAME);
167         // Get Jetty server port
168         HttpListener[] listeners = jettyServer.getListeners();
169         int		impromptuPort = listeners[0].getPort();
170         // get the security hanlder through this awkward communication
171         sh = (LevelHandler) filterConfig.getServletContext().getAttribute(
172                 LevelHandler.ATTRIBUTE_NAME);
173         realm = (SlideUserRealm)filterConfig.getServletContext().getAttribute
174         			(SlideUserRealm.ATTRIBUTE_NAME);
175         // Slide: Dynamicallly add a new user for the current user
176         token = (NamespaceAccessToken)filterConfig.getServletContext().
177 						getAttribute(WebdavServlet.ATTRIBUTE_NAME);
178 
179         // Here we create a special SlideToken. This is the same type of token
180         // 		that the system uses during initialization.
181         // Normally the token should be constructued from the httpRequest, 
182         // but our setting ACL permissions at the server side is triggered through 
183         // 		the event channel, which provides no httpRequest we can leverage.
184         // Again, this relies on the assumption that the event channel is trustworthy
185         // A normal ACL method from the client side (as our first implementation) 
186         //		might be the better choice
187         if (!edu.uci.isr.impromptu.common.Configuration.ACL_AT_CLIENT) {
188 	        SlideToken slideToken = new SlideTokenWrapper(new SlideTokenImpl(new CredentialsToken("")));
189 	        slideToken.setForceSecurity(false);
190 	        slideToken.setForceStoreEnlistment(true);
191 	        aclManager = new ACLManager(token, slideToken);
192         }
193         passwordManager = new PasswordManager();
194 
195         // Now repository and pie are interesetd in each others' events
196         interestedServices.add(Configuration.IMPROMPTU_SERVICE_TYPE);
197         interestedServices.add(edu.uci.isr.impromptu.pie.Configuration.PIE_SERVICE_TYPE);
198         // Use JmDNS to announce the public key
199         Hashtable		properties = new Hashtable();
200         properties.put(PasswordManager.PUBLIC_KEY_FIRST, 
201                 passwordManager.getMyPublicKeyBytesFirst());
202         properties.put(PasswordManager.PUBLIC_KEY_SECOND, 
203                 passwordManager.getMyPublicKeyBytesSecond());
204         // ISSUE: architecture: this really belongs to the Pie service info
205         byte[]	color = edu.uci.isr.impromptu.pie.Configuration.getPieColorBytes(); 
206         if (color != null) {
207             properties.put(edu.uci.isr.impromptu.pie.Configuration.ATTR_PIE_COLOR, color);
208         }
209         myServiceInfo = new ServiceInfo(Configuration.IMPROMPTU_SERVICE_TYPE,
210                 getName() + getCharacter(), impromptuPort, 0, 0, 
211                 properties);
212 
213         // Start the pie before the peer manager,
214         //		because the peer manager might call the pie
215         if (integratePie) {
216             Pie.mainSwing(new String[] {});
217         }
218         pm = new PeerManager(this, 
219                 edu.uci.isr.impromptu.common.Configuration.YANCEES_CONFIGURATION_FILE);
220         pm.registerJmDNS();
221         if (integratePie) {
222             Pie.thePie.setPeerManager(pm);
223         }
224         
225         // responds to events from the pocket pc, that inquires for the list of 
226         // peers and their files.
227         modelServer = new PeerQueryServer(pm);
228         
229

306     /***
307      * Create a peer when given some service info
308      * 
309      * @param si
310      *            the service info
311      * @return a newly created peer
312      */
313     public synchronized Peer createAPeer(ServiceInfo si) {
314         String	serviceType = si.getType();
315         if (!serviceType.equalsIgnoreCase(Configuration.IMPROMPTU_SERVICE_TYPE))
316             return null;
317         
318         if (alwaysAllowOthers) {
319             // TODO: security: architecture: make admission control an aspect
320             // If we use the insecure BASIC, we can get the remote password and put it there
321             // Since we are using more secure DIGEST, we cannot recover 
322             //		password from remote credentials and we have to use our own default
323             // TODO: security: procotol: a better password exchange? BASIC over SSL?
324             // TODO: security: feature: can we tell a research story for asking user to addition? 
325         }
326         String	peerCharacterName = si.getName();
327         String  peerName = Utils.getPeerUserName(peerCharacterName);
328         String  roleName = peerName.equals(localUserName)? SecurityConstants.ME :
329             						SecurityConstants.OTHERS;
330         // Add to Jetty
331         byte[] 	publicKey = getPublicKeyBytes(si);
332         // TODO: security: protocol: now it is a big problem. 1) This password is only valid for 
333         //		one pair. So it is not simplly proxyable (where the client has only 
334         //		one password (the proxy password). The proxy can of course host 
335         //		many passwords. 2) Is there some sort of group DH mechanism? 
336         //		3) How could this be single sign on?
337         log.debug(localUserName + " adds " + peerName);
338         String	password = passwordManager.getSecretPassword(publicKey);
339         if (roleName.equals(SecurityConstants.ME))
340             // TODO: security: architecture: this is to ease test
341             realm.addUser(peerName, "password", roleName);
342         else
343             realm.addUser(peerName, password, roleName);
344         // Add to Slide
345         UserManager.addASlideUserAtServer(token, peerName, roleName);
346         // ISSUE: security: context: remove users
347 
348         // user our own name when authenticate to the peer (whose name is peer name)
349         RepositoryPeer p = new RepositoryPeer(si, localUserName, password);
350         if (roleName.equals(SecurityConstants.ME))
351             localPeer = p;
352         return p;
353     }

355     // Currently a slice peer is created after a repository peer.
356     // During creating a slice peer, it pulls its files from the proxy. 
357     // (A possible bypass is to pull directly from the repository peer, but that is against
358     // the central theme of doing everything through the proxy. [Humh, a trusted mobile code
359     // becomes more attractive at this point.]) Pulling from the 
360     // proxy is possible only after the password (between the Pie client and the 
361     // proxy) is set. The setting of password happens during the creation of the 
362     // local repository peer. However, JmDNS does not assure that the local peer is the 
363     // first to create. So we must save other peers until the local peer is created. 
364     private List otherSlicesToCreate = new ArrayList();
365     /***
366      * Perform further initialization after the peer is created and added to 
367      * the peer manager
368      * 
369      * @param si the service info of the newly appeared peer
370      */
371     public void	afterCreateAPeer(final ServiceInfo si){
372         String	serviceType = si.getType();
373         if (!serviceType.equalsIgnoreCase(Configuration.IMPROMPTU_SERVICE_TYPE))
374             return;
375 
376         // is the name of the peer that was added.
377         final String  newPeerName = si.getName();
378         
379         final String  roleName = newPeerName.equals(localUserName)? SecurityConstants.ME :
380             						SecurityConstants.OTHERS;
381 
382         // announce the added "others" peer
383         // This should be put here since Yancees is added after createAPeer in PeerManager
384         new Thread() {
385             public void run() {
386                 try {
387                     // This extra waiting is to compensate Yancees thread during its setup
388                     // Another solution is to wait for the inquiry before sending this message.
389                     //	 But that complicates the protocol without much advantage. 
390                     Thread.sleep(2000);
391                 }
392                 catch(InterruptedException ie) {
393                 }
394 		        if (!roleName.equals(SecurityConstants.ME)) {
395 			        Map addedEvent = new HashMap();
396 			        addedEvent.put(MessageProcessor.ATTR_PEER_NAME, localUserName);
397 			        addedEvent.put(MessageProcessor.ATTR_ACTOR, newPeerName);
398 			        addedEvent.put(MessageProcessor.ATTR_ACTION, MessageProcessor.ACTION_ADD_PEER);
399 			        
400 			        Color color = PieUtils.getColorFromServiceInfo(si);
401 			        if (color == null) {
402 			        	color = PieUtils.getColorForName(newPeerName);
403 			        }
404 			        	        
405 			        addedEvent.put(MessageProcessor.ATTR_COLOR, PieUtils.getRGBString(color));
406 			        
407 			        pm.sendMessage(addedEvent);
408 		        }
409             }
410        }.start();
411 
412         byte[] 	publicKey = getPublicKeyBytes(si);
413         String	password = passwordManager.getSecretPassword(publicKey);
414         if (integratePie) {
415             if (localPeer==null) {
416                 // wait until the local peer is up to create other slices
417                 otherSlicesToCreate.add(si);
418                 return;
419             }
420 
421             // let the pie create its peer
422             if (roleName.equals(SecurityConstants.ME)) {
423                 // Here the repository passes something to the pie. 
424                 //		This is a rather tight coupling. 
425                 // Securitywise, we are generating a password for each peer, even 
426                 //		among ourselves, so this might be necessary.
427                 // This is also the reason for this strange-looking afterCreateAPeer
428                 //		method. During setPassword, we mount the drive (we have 
429                 //		the username and password; if we wait, then until when?)
430                 //		However, if this was in creaetAPeer method, the peer manager
431                 //		has not had a chance to add this peer into its set, so
432                 //		in doFilter's pm.getPeers, it returns a wrong result. 
433                 //		This is the situation when the net use command reaches
434                 //		the doFilter, because the repository peer has not been 
435                 //		added yet. Thus we add this method and let peer manager
436                 //		call it one more time after it has added the peer.
437                 // TODO: security: architecture: this is to ease test
438                 if (!password.equals("password")) { 
439                     Pie.thePie.setPassword("password");
440                     localPassword = "password";
441                 }
442                 //Pie.thePie.setPassword(password);
443                 //localPassword = password;
444 
445                 // pick up any update requests that were sent before we are up
446                 for (Iterator i = otherSlicesToCreate.iterator(); i.hasNext(); ) {
447                     ServiceInfo otherSi = (ServiceInfo)i.next();
448                     Pie.thePie.createAPeer(otherSi);
449                 }
450                 otherSlicesToCreate.clear();
451             }
452             // This needs to be after setPassword, since the pie pulls files
453             //		during createAPeer
454             Pie.thePie.createAPeer(si);
455         }
456         else {
457             // Print the password on console so a separate pie can authenticate
458             System.out.println("The password is " + password);
459         }
460     }

464     private static byte[] getPublicKeyBytes(ServiceInfo si) {
465         byte[]	first = si.getPropertyBytes(PasswordManager.PUBLIC_KEY_FIRST);
466         byte[]	second = si.getPropertyBytes(PasswordManager.PUBLIC_KEY_SECOND);
467         byte[]	publicKeyBytes = new byte[first.length+second.length];
468         System.arraycopy(first, 0, publicKeyBytes, 0, first.length);
469         System.arraycopy(second, 0, publicKeyBytes, first.length, second.length);
470         return	publicKeyBytes;
471     }

498     /***
499      * direct the request to an appropriate peer
500      */
501     public void doFilter(ServletRequest request, ServletResponse response,
502             FilterChain chain) throws IOException, ServletException {
503         if (filterConfig == null)
504             return;
505 
506         if (!(request instanceof HttpServletRequest)
507                 || !(response instanceof HttpServletResponse))
508             return;
509         
510         if (localPeer == null) {
511             log.error("Empty local peer");
512             return;
513         }
514 
515         HttpServletRequest 		hRequest = (HttpServletRequest) request;
516         HttpServletResponse 	hResponse = (HttpServletResponse) response;
517         CharResponseWrapper 	wrapper = new CharResponseWrapper(hResponse);
518         String 					method = hRequest.getMethod();
519         String 					filename = RepositoryUtils.getFileName(hRequest);
520         Peer 					owner = pm.getOwnerPeer(filename);
521 
522         if (hRequest.getHeader(Configuration.IMPROMPTU_HEADER_NAME) != null) {
523             // the request is from an Impromptu that does not want us to propgate
524             // this should not be used as an authentication mechanism (because 
525             // others can fake the header)?
526             doFilterAndStatus(hRequest, hResponse, wrapper, chain);
527         }
528         else if (filename != null && Utils.isRoot(filename)
529                 && method.equals(WebdavConstants.M_PROPFIND)) {
530             // Must save the http request in a request wrapper, because the
531             // request goes to ourselves and other peers, and getInputStream can only
532             // be called on one request once
533             BufferedRequestWrapper bufferedRequest = 
534                 						new BufferedRequestWrapper(hRequest);
535 
536             // if it is a propfind on the root path, we need to aggretate peers' responses
537             // This might become a bottle neck. What can be done is to cache results
538             // and use events to invalidate
539 
540             // the responses from different peers
541             // make a copy of peer manager's peers, because it will change
542             Map peers = new HashMap(pm.getPeers());
543             // Keep return response document
544             Document[] responses = new Document[peers.size()];
545             // Analyze and initialize a peer's file list, if necessary
546             RepositoryPeer[]	rps = new RepositoryPeer[peers.size()];	
547 
548             // first get the response from myself
549             chain.doFilter(bufferedRequest, wrapper);
550             if (wrapper.getStatus() >= WebdavStatus.SC_BAD_REQUEST) {
551                 responses[0] = null;
552                 System.err.println(wrapper.getStatusText());
553             }
554             else {
555                 try {
556                     synchronized (documentBuilder) {
557                         // SAX parse is not thread safe
558 	                    responses[0] = documentBuilder.parse(new ByteArrayInputStream(
559 	                            				wrapper.toString().getBytes()));
560                     }
561                     rps[0] = localPeer;
562                 }
563                 catch (SAXException e) {
564                     Utils.handleException(e);
565                 }
566                 catch (IOException e) {
567                     Utils.handleException(e);
568                 }
569             }
570 
571             // then collect responses from others. slow operation
572             Iterator itr = peers.values().iterator();
573             int i = 1;
574             while (itr.hasNext()) {
575                 Object o = itr.next();
576                 if (o instanceof RepositoryPeer) {
577                     RepositoryPeer p = (RepositoryPeer) o;
578                     // ISSUE: security: naming: one more name based direction, is it secure?
579                     // TODO: security: delegate: this is the proxy, how does this affect delegate?
580                     // TODO: security: delegate: EJB has runas for servlet spec,
581                     //			can this be made to a delegate connector?
582                     // ip-based connector vs secret-exchange 
583                     // 		connector, and other connectors. 
584                     if (p != localPeer) {
585                         responses[i] = (Document) p.service(bufferedRequest,
586                                 hResponse, false, true);
587                         rps[i] = p;
588                         i++;
589                     }
590                 }
591             }
592 
593             constructResponse(hRequest, responses, rps, hResponse);
594         }
595         // ISSUE: security: naming: here is a name based direction, is it secure?
596         else if (owner != null && owner instanceof RepositoryPeer
597                 && owner != localPeer) {
598             RepositoryPeer	rp = (RepositoryPeer)owner;
599             // the owner is not the local owner
600             if (method.equalsIgnoreCase(WebdavConstants.M_MOVE)) {
601                 // We are MOVEing a file that is now owned by us to a local file
602                 //
603                 // On Mac, and also with Office 2000, when an application saves 
604                 //		a file, it creates a temporary file, saves the changes to
605                 //		that file, and rename the temporary file to the original file.
606                 // This creates an issue for the file identity, which is the basis
607                 //		of our proxy. The application issues a MOVE request to move
608                 //		a file (owned by another peer) to a temporary file (created and
609                 //		owned by the local peer). This change of identity and ownership
610                 //		is not allowed by our design. 
611                 // The work around here is 1) when an application on local peer tries to
612                 //		MOVE a file owned by another peer to a local file (the appcliation
613                 //		always think it is dealing with local files), the original peer
614                 //		suspends its ownership of the file, the local peer grabs
615                 //		the ownership, and make a local copy of the resource. 2) when
616                 //		the application tries to MOVE the file back, the local peer releases
617                 //		the ownership, the remote peer resumes the ownership, and the changed
618                 //		copy is PUT back to the original location.
619 
620                 // create a local copy so future requests go to this copy
621                 try {
622                     // can change the level for other peers' file thus always challenges
623                     // This can be a more mobile code story, possibley with SWT OLE
624                     WebdavResource remote = new WebdavResource(new HttpURL(
625                             rp.getRemoteUserName(), 
626                             rp.getPassword(),  
627                             rp.getHostAddress(), 
628                             rp.getPort(), 
629                             Utils.fileNameToURL(filename)), 
630                             WebdavResource.NOACTION, DepthSupport.DEPTH_0);
631         	        remote.retrieveSessionInstance().getState().setAuthenticationPreemptive(false);
632                     String	data = remote.getMethodDataAsString();
633                     // transfer ownership from the original owner to the current owner
634                     rp.suspendFile(filename, localPeer);
635                     sh.setSecurityLevelForOthers(filename, SecurityConstants.LEVEL_EXISTENCE);
636                     WebdavResource local =  new WebdavResource(new HttpURL(
637                             localUserName, localPassword, localPeer.getHostAddress(), 
638                             localPeer.getPort(), 
639                             Utils.fileNameToURL(filename)),

641         	        local.retrieveSessionInstance().getState().setAuthenticationPreemptive(false);
642                     // When we issue the following methods towards the local proxy, 
643                     //		it goes locally, since the we have grabbed the ownership
644                     local.lockMethod();
645                     local.putMethod(data);
646                     local.unlockMethod();
647                 }
648                 catch (Exception e) {
649                     Utils.handleException(e);
650                 }
651                 // redirect it back to ourselves, towards the local copy just made
652                 doFilterAndStatus(hRequest, hResponse, wrapper, chain);

654                 // The following comments are for moving in the original server
655                 // 	Move the original file to a new name, can only be done 
656                 //		within a server. The headerValue will be relative 
657                 //		to the proxy. It should be changed to the residing host
658                 // One more instance that needs setRequestHeader, not addRequestHeader
659             }
660             else {
661                 rp.service(hRequest, hResponse, true, false);
662             }
663             // fake slide_uri for the LogFilter in case of bypassing
664             if (hRequest.getAttribute("slide_uri") == null)
665                 hRequest.setAttribute("slide_uri", filename);
666         }
667         else {
668             // it should go to the local peer
669             doFilterAndStatus(hRequest, hResponse, wrapper, chain);
670             if (method.equalsIgnoreCase(WebdavConstants.M_MOVE)) {  
671                 String	newFileName = RepositoryUtils.getFileName(
672 	                	new HttpURL(hRequest.getHeader("Destination")).getPath());
673                 RepositoryPeer originalOwner = localPeer.isGrabbed(newFileName);
674                 if (originalOwner != null) {
675                     // The local application has just moved a temporary file back to 
676                     //		the original file
677                     // The temporary file might not even be the one named in the 
678                     //		original MOVE request, since it can create and move
679                     //		many intermediate files
680                     // Get the original owner
681 	                // send back changes
682                     try {
683                         WebdavResource local =  new WebdavResource(new HttpURL(
684                                 localUserName, localPassword, localPeer.getHostAddress(), 
685                                 localPeer.getPort(), 
686                                 Utils.fileNameToURL(newFileName)), 
687                                 WebdavResource.NOACTION, DepthSupport.DEPTH_0);
688             	        local.retrieveSessionInstance().getState().setAuthenticationPreemptive(false);
689                         String	data = local.getMethodDataAsString();
690                         localPeer.releaseFile(newFileName);
691                         // resume ownership so future changes can be redirected back 
692                         //		to the original owner
693                         originalOwner.resumeFile(newFileName);
694                         // No readd event here, since we do not change the identity 
695                         //		of the original file that is owned by another
696                         WebdavResource remote = new WebdavResource(new HttpURL(
697                                 originalOwner.getRemoteUserName(), 
698                                 originalOwner.getPassword(),  
699                                 originalOwner.getHostAddress(), 
700                                 originalOwner.getPort(), 
701                                 Utils.fileNameToURL(newFileName)), 
702                                 WebdavResource.NOACTION, DepthSupport.DEPTH_0);
703             	        remote.retrieveSessionInstance().getState().setAuthenticationPreemptive(false);
704                         remote.lockMethod();
705                         remote.putMethod(data);
706                         remote.unlockMethod();
707                     }
708                     catch (Exception e) {
709                         Utils.handleException(e);
710                     }
711                 }
712             }
713        }

747                 String		entryURI = "";

769                             // SEE_ONLY is a good start.
770                             // It does not matter much, since only the existence of

774                     }

779                             if (!localPeer.isShared(entryName)) {
780                                 include = false;

784                     if (include)

786                                 peerResponse, true));

793                 .getOutputStream());

801     //			listeners

807     /***
808      * Execute the request locally, and possibly announce events
809      * 
810      * @param request
811      *            the request
812      * @param hResponse	
813      * 			  the original response that will receive our output
814      * @param wrapper
815      *            the response with status and content
816      * @param chain
817      *            the filter chain
818      * @throws IOException
819      * @throws ServletException
820      */
821     private void doFilterAndStatus(HttpServletRequest request, 
822             HttpServletResponse hResponse, CharResponseWrapper wrapper, FilterChain chain)
823             throws IOException, ServletException {
824         String 		method = request.getMethod();
825         String 		fileName = RepositoryUtils.getFileName(request);
826         boolean		doingDir = false;
827         if (fileName != null && Utils.isRoot(fileName) && 
828             method.equals(WebdavConstants.M_PROPFIND)) {
829             	doingDir = true;
830         }
831         StatusHttpServletResponseWrapper	response;
832         if (doingDir) {
833             response = wrapper;
834         }
835         else {
836             // if we are not handling PROPFIND, we just need a status, 
837             // no need to buffer the content
838             response = new StatusHttpServletResponseWrapper(hResponse);
839         }
840 
841         // do filter chain
842         chain.doFilter(request, response);
843 
844         // check status of method
845         int status = response.getStatus();
846         if (doingDir) {
847 		    InputStream in = new ByteArrayInputStream(response.toString().getBytes());
848             if (status == WebdavStatus.SC_OK || status == WebdavStatus.SC_MULTI_STATUS) {
849 	            // we have a response, do hiding invisible files
850 	            Document[]	responses = new Document[1];
851 	            RepositoryPeer[] rps = {localPeer};
852 	            try {
853 	                synchronized(documentBuilder) {
854 	                    responses[0] = documentBuilder.parse(in);
855 	                }
856 	            }
857 	            catch (SAXException e) {
858 	                Utils.handleException(e);
859 	            }
860 	            catch (IOException e) {
861 	                Utils.handleException(e);
862 	            }
863 	            constructResponse(request, responses, rps, hResponse);
864             }
865             else {
866                 // other headers should have been set
867             	// copy status 
868                 // also copy status text so LogFilter can have the status 
869                 hResponse.sendError(status, response.getStatusText());
870 
871                 // copy body
872     		    OutputStream out = hResponse.getOutputStream();
873     		    byte[] buffer = new byte[2048];
874     		    int len = 0;
875     		    if(in != null) {
876     		        while((len = in.read(buffer, 0, buffer.length)) != -1) {
877     		            out.write(buffer, 0, len);
878     		        }
879     		        in.close();
880     		    }
881     		    out.flush();
882     		    out.close();
883             }
884         }
885         else if (method.equals(WebdavConstants.M_PUT)
886                 && status == WebdavStatus.SC_CREATED) {
887             // Not to broadcast hidden files (private/pureExistence). Hide them in PROPFIND. 
888             //  This needs the same naming/caching of file names.
889             //  The bundle architecture requires we broadcast a hidden file
890             //	since our pie needs the information to create something to drag.
891             //  Two solutions 1) architecting for private connectors 2) encrypt
892             //		the file information so only me can decrypt.
893             //
894             // Grabing a name during first put is important, but might be 
895             //	secondary to security concerns. Don't over complicate this.
896             /* No longer generate events on random file creations. 
897              * Now must use explicit drop in Pie to generate addedEvent.
898              * Originally there were two choices: server control security (here)
899              * and client control security (pie proppatch)
900             Map addedEvent = new HashMap();
901             addedEvent.put(MessageProcessor.ATTR_PEER_NAME, userName);
902             addedEvent.put(MessageProcessor.ATTR_FILE_NAME, fileName);
903             addedEvent.put(MessageProcessor.ATTR_ACTION,
904                     MessageProcessor.ACTION_ADD);
905             pm.sendMessage(addedEvent);
906             */
907             // Set default security for any files created. This servers two purposes.
908             // 1) So others cannot peek them. (These files are not even private, because
909             // private files will show up in the pie. These files only show up in
910             // a local PROPFIND. ) Even others know their existence the permissons
911             // are still secure by default. The original assumption that only we create
912             // files in the repository is wrong, since there are files users do not 
913             // explicitly create. Again, an integration issue.
914             // 2) So the Jetty can send authentication challenges to ME. Without protection
915             // Jetty will let an unauthenticated user in, which will reject by Slide.
916             sh.setSecurityLevelForOthers(fileName, SecurityConstants.LEVEL_EXISTENCE);
917         }
918         else if (method.equals(WebdavConstants.M_MOVE)
919                 && status == WebdavStatus.SC_CREATED) {
920             String	newFileName = RepositoryUtils.getFileName(
921                 	new HttpURL(request.getHeader("Destination")).getPath());
922             // set security for files thus created
923             if (localPeer.isPrivate(newFileName) || localPeer.isShared(newFileName)) {
924 		        // a local file is readded, and thus all its properties and 
925                 //	security settings have been lost. Its original owner 
926                 //	should re-proppatch
927 		        Map readdedEvent = new HashMap();
928 		        readdedEvent.put(MessageProcessor.ATTR_PEER_NAME, localUserName);
929 		        readdedEvent.put(MessageProcessor.ATTR_FILE_NAME, newFileName);
930 		        readdedEvent.put(MessageProcessor.ATTR_ACTION,
931 		                MessageProcessor.ACTION_RE_ADD);
932 		        pm.sendMessage(readdedEvent);
933             }
934             else {
935                 sh.setSecurityLevelForOthers(newFileName, SecurityConstants.LEVEL_EXISTENCE);
936             }
937         }
938         else if (localPeer.isShared(fileName)){
939             // only send messages for shared files
940 	        if (method.equals(WebdavConstants.M_DELETE)
941 	                && (status == WebdavStatus.SC_OK || status == WebdavStatus.SC_NO_CONTENT)) {
942 	            // a file is removed
943 	            Map deletedEvent = new HashMap();
944 	            deletedEvent.put(MessageProcessor.ATTR_PEER_NAME, localUserName);
945 	            deletedEvent.put(MessageProcessor.ATTR_FILE_NAME, fileName);
946 	            deletedEvent.put(MessageProcessor.ATTR_ACTION,
947 	                    MessageProcessor.ACTION_REMOVE);
948 	            pm.sendMessage(deletedEvent);
949 	        }
950 	        else if (method.equals(WebdavConstants.M_PUT)
951 	                && (status == WebdavStatus.SC_OK || status == WebdavStatus.SC_NO_CONTENT)) {
952 	            //if (!request.getRemoteUser().equals(localUserName)) {
953 	            // a file is modified by somebody
954 	            Map writeEvent = new HashMap();
955 	            writeEvent.put(MessageProcessor.ATTR_PEER_NAME, localUserName);
956 	            writeEvent.put(MessageProcessor.ATTR_FILE_NAME, fileName);
957 	            writeEvent.put(MessageProcessor.ATTR_ACTION,
958 	                    MessageProcessor.ACTION_WRITE);
959 	            writeEvent.put(MessageProcessor.ATTR_ACTOR,
960 	                    request.getRemoteUser());
961 	            pm.sendMessage(writeEvent);
962 	            //}
963 	        }
964 	        else if (method.equals(WebdavConstants.M_GET)
965 	                && (status == WebdavStatus.SC_OK)) {
966 	            if (!request.getRemoteUser().equals(localUserName)) {
967 	                // a file is read by somebody
968 		            Map readEvent = new HashMap();
969 		            readEvent.put(MessageProcessor.ATTR_PEER_NAME, localUserName);
970 		            readEvent.put(MessageProcessor.ATTR_FILE_NAME, fileName);
971 		            readEvent.put(MessageProcessor.ATTR_ACTION,
972 		                    MessageProcessor.ACTION_READ);
973 		            readEvent.put(MessageProcessor.ATTR_ACTOR,
974 		                    request.getRemoteUser());
975 		            pm.sendMessage(readEvent);
976 	            }
977 	        }
978         }

982      * Process the message

986     public void processMessage(Map message) {
987         String owner = (String) message.get(MessageProcessor.ATTR_PEER_NAME);
988         if (owner != null) {
989             Peer ownerPeer = pm.getPeer(owner);
990             if (ownerPeer != null) {
991                 // ISSUE: init: suppose we missed a JmDNS annoucement so it is null?
992                 String fileName = (String) message.get(MessageProcessor.ATTR_FILE_NAME);
993                 String action = (String) message.get(MessageProcessor.ATTR_ACTION);
994                 String radius = (String) message.get(MessageProcessor.ATTR_RADIUS);
995                 String angle = (String) message.get(MessageProcessor.ATTR_ANGLE);
996                 String level = (String) message.get(MessageProcessor.ATTR_LEVEL);
997                 double r = 0;
998                 double a = 0;
999                 int l = SecurityConstants.LEVEL_PRIVATE;
1000                 try {
1001                     r = Double.parseDouble(radius);
1002                     a = Double.parseDouble(angle);
1003                     l = Integer.parseInt(level);
1004                 }
1005                 catch (Exception e) {
1006                 }
1007                 if (action.equals(MessageProcessor.ACTION_ADD)) {
1008                     ownerPeer.fileAdded(fileName, r, a);
1009                     // only set security if we are the owner
1010                     if (ownerPeer == localPeer) {
1011                         sh.setSecurityLevelForOthers(fileName, l);
1012                         if (!edu.uci.isr.impromptu.common.Configuration.ACL_AT_CLIENT) {
1013 	                        //aclManager.setSecurityLevel(fileName, SecurityConstants.ME, 
1014 	                        //        SecurityConstants.LEVEL_PERSISTENT);
1015 	                        aclManager.setSecurityLevel(fileName, SecurityConstants.OTHERS, l);
1016                         }
1017                     }
1018                 }
1019                 else if (action.equals(MessageProcessor.ACTION_MOVE)) {
1020                     ownerPeer.fileMoved(fileName, r, a);
1021                     if (ownerPeer == localPeer) {
1022                         sh.setSecurityLevelForOthers(fileName, l);
1023                         if (!edu.uci.isr.impromptu.common.Configuration.ACL_AT_CLIENT) {
1024 	                        //aclManager.setSecurityLevel(fileName, SecurityConstants.ME, 
1025 	                        //        SecurityConstants.LEVEL_PERSISTENT);
1026 	                        aclManager.setSecurityLevel(fileName, SecurityConstants.OTHERS, l);
1027                         }
1028                     }
1029                 }
1030                 else if (action.equals(MessageProcessor.ACTION_REMOVE)) {
1031                     ownerPeer.fileRemoved(fileName);
1032                 }
1033                 else if (action.equals(MessageProcessor.ACTION_SHUTDOWN)) {
1034                     // ISSUE: security: architecture: here we use message to notify shutdown.
1035                     // Jetty uses a stop.jar that employs a protected port. Which is better?
1036                     // Also, we shutdown here since the server starts on this side. How about Pie shutdown?
1037                     if (ownerPeer == localPeer) {
1038                         // Shutdown ourselves
1039                         try {
1040                             jettyServer.stop();
1041                             jettyServer.destroy();
1042                             System.exit(0);
1043                         }
1044                         catch (InterruptedException e) {
1045                             Utils.handleException(e);
1046                         }
1047                     }
1048                 }
1049             }
1050         }
1051         if (integratePie) {
1052             //  Here the pie always processes messages 
1053             //	after the proxy. But in an independent world, this is not
1054             //	the case. There will be no such garantee, and thus raises
1055             //	a potential timing problem.
1056             Pie.thePie.processMessage(message);
1057         }
1058     }
